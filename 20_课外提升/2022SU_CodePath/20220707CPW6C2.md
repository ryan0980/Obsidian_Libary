class MovingAverage {
 
    queue<int> numbers;
    int windowSize = 0;
    double currentAvg = 0; 
    //maybe current sum?

    public MovingAverage(int size) {
         windowSize = size;
    }
    
    public double next(int val) {
        numbers.push(val);
        
        if(numbers.size() > windowSize){ //10 3 5
            numbers.pop();
        } 
    
        currentAvg = 0;
        for(int i = 0; i < numbers.size(); i++){
            int currVal = numbers.top();
            currentAvg+=currVal;
            numbers.pop();
            numbers.push(currVal);
        }
        currentAvg/=windowSize;
    
        return currentAvg;
    }
}
